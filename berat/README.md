# how to prepare
1. setting env
    DB_CONNECTION=mysql
    DB_HOST=127.0.0.1
    DB_PORT=3306
    DB_DATABASE=name_db
    DB_USERNAME=username
    DB_PASSWORD=password

2. RUN php artisan migrate
3. RUN php artisan db:seed

# how to running
1. php artisan serve
2. running in localhost:yourport/berat
3. automatic testing with selenium ->> path folder testing-selenium
